# doc2md.py Python dependencies
#
# pdfmux: PDF -> Markdown with per-page self-healing extraction + confidence
#         scoring. Requires Python 3.11+.
#
#         The extras below give the local router more backends to choose from,
#         which is the "maximum local effort" doc2md uses by default (no cloud
#         LLM unless you pass --llm):
#           [ocr]    RapidOCR for the occasional scanned page
#           [tables] Docling for high-fidelity table extraction
#         Add [marker]/[opendataloader] too if your corpus has academic papers
#         or very complex multi-column layouts; or use pdfmux[all] for everything.
pdfmux[ocr,tables]>=1.7.0

# PyMuPDF: drives the figure/table extraction pass (rendering raster images and
# complex tables to PNG) and --preview page slicing. Also a pdfmux dependency,
# listed explicitly because doc2md.py imports it directly.
pymupdf>=1.24.0

# Optional, only needed for --yaml (XML -> YAML output). Imported lazily, so
# users who never pass --yaml don't need these installed.
#   xmltodict: XML -> dict (attributes -> @name, text -> #text, repeats -> list)
#   PyYAML:    dict -> YAML
xmltodict>=0.13.0
PyYAML>=6.0
